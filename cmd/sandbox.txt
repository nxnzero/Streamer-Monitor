package main

import (
	"LycorisFlower/internal/trovo"
	"bytes"
	"encoding/json"
	"fmt"
	"net/http"
	"os"
)

func main() {
	client := &http.Client{}

	data := trovo.RefreshTokenRequest{
		ClientSecret: os.Getenv("TROVO_CUSTOMER_SECRET"),
		GrantType:    "refresh_token",
		RefreshToken: os.Getenv("TROVO_REFRESH_TOKEN"),
	}

	dataJSON, err := json.Marshal(data)
	if err != nil {
		fmt.Println(err)
	}

	request, err := http.NewRequest("POST", trovo.URL_TOKEN_REFRESH, bytes.NewBuffer(dataJSON))
	if err != nil {
		fmt.Println(err)
	}

	request.Header.Set("Content-Type", "application/json")
	request.Header.Set("Accept", "application/json")
	request.Header.Set("client-id", os.Getenv("TROVO_CLIENT_ID"))

	response, err := client.Do(request)
	if err != nil {
		fmt.Println(err)
	}

	defer response.Body.Close()

	var responseJSON trovo.ResponseTokenRequest

	if err := json.NewDecoder(response.Body).Decode(&responseJSON); err != nil {
		fmt.Println(err)
	}

	fmt.Printf("Access: %v\nRefresh: %v\n",
		responseJSON.AccessToken,
		responseJSON.RefreshToken)
}
